void FlashPhotoApp::applyFilterSharpen()
{
cout << "Apply has been clicked for Sharpen with amount = " << m_filterParameters.sharpen_amount << endl;
float sharpen = m_filterParameters.sharpen_amount;
float Kernel[3][3] = {{-sharpen/8, -sharpen/8, -sharpen/8},{-sharpen/8, sharpen + 1, -sharpen/8},{-sharpen/8, -sharpen/8, -sharpen/8}};
m_displayBuffer->applyConvolution(Kernel);
}
