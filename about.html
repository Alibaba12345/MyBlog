void FlashPhotoApp::applyFilterSpecial() {
cout << "Apply has been clicked for Special" << endl;
int width = m_displayBuffer->getWidth();
int height = m_displayBuffer->getHeight();
for (int x = 0; x < width; x++)
{
for (int y = 0; y < height; y++)
{
ColorData oldC = m_displayBuffer->getPixel(x, y);
ColorData newC = ColorData(floor(oldC.getRed()), floor(oldC.getGreen()), floor(oldC.getBlue()));
ColorData errC = oldC - newC;
m_displayBuffer->setPixel(x, y, newC);
m_displayBuffer->setPixel(x+1, y, m_displayBuffer->getPixel(x+1, y) + errC * (7/16));
m_displayBuffer->setPixel(x-1, y+1, m_displayBuffer->getPixel(x-1, y+1) + errC * (3/16));
m_displayBuffer->setPixel(x, y+1, m_displayBuffer->getPixel(x, y+1) + errC * (5/16));
m_displayBuffer->setPixel(x+1, y+1, m_displayBuffer->getPixel(x+1, y+1) + errC * (1/16));
}
}
}
